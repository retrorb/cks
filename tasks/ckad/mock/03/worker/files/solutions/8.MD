# Create overlay directory
mkdir -p /home/ubuntu/kustomize/overlays/prod

# vim /home/ubuntu/kustomize/overlays/prod/kustomization.yaml

namespace: kust-ns
resources:
- ../../base
patches:
- patch: |
    - op: replace
      path: /spec/replicas
      value: 3
  target:
    kind: Deployment
    name: kust-app
```
```
# Apply the overlay
kubectl apply -k /home/ubuntu/kustomize/overlays/prod

# Verify
kubectl get deployment -n kust-ns
kubectl get pods -n kust-ns
