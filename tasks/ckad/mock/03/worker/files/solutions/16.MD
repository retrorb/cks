```
# vim 16.yaml

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-frontend
  namespace: netpol-ns
spec:
  podSelector:
    matchLabels:
      role: backend
  policyTypes:
  - Ingress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          role: frontend
    ports:
    - protocol: TCP
      port: 8080
```
```
kubectl apply -f 16.yaml

# Verify
kubectl get networkpolicy allow-frontend -n netpol-ns

# Test - frontend can reach backend
backend_ip=$(kubectl get pod backend -n netpol-ns -o jsonpath='{.status.podIP}')
kubectl exec frontend -n netpol-ns -- wget -q --timeout=5 -O- "http://$backend_ip:8080"

# Test - other cannot reach backend
kubectl exec other -n netpol-ns -- wget -q --timeout=5 -O- "http://$backend_ip:8080"
```
