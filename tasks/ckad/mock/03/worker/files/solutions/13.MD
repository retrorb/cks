# vim 13.yaml

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: web-ingress
  namespace: ingress-basic-ns
spec:
  ingressClassName: nginx
  rules:
  - host: web.example.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: web-svc
            port:
              number: 80
```
```
kubectl apply -f 13.yaml

# Verify
kubectl get ingress web-ingress -n ingress-basic-ns
kubectl describe ingress web-ingress -n ingress-basic-ns
