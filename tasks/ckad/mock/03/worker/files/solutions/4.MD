```
# Check current state
kubectl get deployment rollback-app -n rollback-ns
kubectl get pods -n rollback-ns
```
```
# Check rollout history
kubectl rollout history deployment/rollback-app -n rollback-ns
```
```
# Rollback to previous revision
kubectl rollout undo deployment/rollback-app -n rollback-ns

# Verify
kubectl rollout status deployment/rollback-app -n rollback-ns
kubectl get deployment rollback-app -n rollback-ns -o jsonpath='{.spec.template.spec.containers[0].image}'
kubectl get pods -n rollback-ns
```
