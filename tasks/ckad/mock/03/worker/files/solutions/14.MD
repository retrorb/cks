# vim 14.yaml

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: multi-ingress
  namespace: ingress-multi-ns
spec:
  ingressClassName: nginx
  rules:
  - http:
      paths:
      - path: /api
        pathType: Prefix
        backend:
          service:
            name: api-svc
            port:
              number: 8080
      - path: /web
        pathType: Prefix
        backend:
          service:
            name: web-svc
            port:
              number: 80
```
```
kubectl apply -f 14.yaml

# Verify
kubectl get ingress multi-ingress -n ingress-multi-ns
kubectl describe ingress multi-ingress -n ingress-multi-ns
