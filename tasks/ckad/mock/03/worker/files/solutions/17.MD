```
# vim 17.yaml

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-trusted
  namespace: app-ns
spec:
  podSelector: {}
  policyTypes:
  - Ingress
  ingress:
  - from:
    - namespaceSelector:
        matchLabels:
          access: trusted
```
```
kubectl apply -f 17.yaml

# Verify
kubectl get networkpolicy allow-trusted -n app-ns

# Test - pod in trusted-ns can reach app-ns

# Test - pod in untrusted-ns cannot reach app-ns
```
