# vim 1.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: web-app
  namespace: deploy-ns
spec:
  replicas: 3
  selector:
    matchLabels:
      app: web-app
  template:
    metadata:
      labels:
        app: web-app
    spec:
      containers:
      - name: web-app
        image: nginx:alpine
        resources:
          requests:
            cpu: "50m"
            memory: "64Mi"
```
```
kubectl apply -f 1.yaml

kubectl get deployment web-app -n deploy-ns
