# vim db-external.yaml

apiVersion: v1
kind: Service
metadata:
  name: db-external
  namespace: extname-ns
spec:
  type: ExternalName
  externalName: database.example.internal
  ports:
  - port: 5432
    protocol: TCP
```
```
kubectl apply -f db-external.yaml
```
```
# Verify
kubectl get svc db-external -n extname-ns
kubectl get svc db-external -n extname-ns -o yaml

# Pods can now reach the external DB using:
# db-external.extname-ns.svc.cluster.local:5432
