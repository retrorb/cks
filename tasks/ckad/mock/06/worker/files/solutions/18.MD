```
# vim egress-to-backend.yaml

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-client-to-backend
  namespace: client-ns
spec:
  podSelector:
    matchLabels:
      role: client
  policyTypes:
  - Egress
  egress:
  - to:
    - podSelector:
        matchLabels:
          role: backend
    ports:
    - protocol: TCP
      port: 8080
```
```
kubectl apply -f egress-to-backend.yaml
```
```
# Verify
backend_ip=$(kubectl get pod backend-pod -n client-ns -o jsonpath='{.status.podIP}')
external_ip=$(kubectl get pod external-pod -n client-ns -o jsonpath='{.status.podIP}')

# client can reach backend (should succeed)
kubectl exec client-pod -n client-ns -- wget -q --timeout=5 -O- "http://$backend_ip:8080"

# client cannot reach external (should fail)
kubectl exec client-pod -n client-ns -- wget -q --timeout=5 -O- "http://$external_ip:8080"
```
