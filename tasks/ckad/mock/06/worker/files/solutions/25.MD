```
# vim e2e.yaml

apiVersion: v1
kind: Service
metadata:
  name: full-svc
  namespace: e2e-ns
spec:
  type: ClusterIP
  selector:
    app: full-app
  ports:
  - port: 80
    targetPort: 8080
---
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: full-ingress
  namespace: e2e-ns
spec:
  ingressClassName: nginx
  rules:
  - host: full.example.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: full-svc
            port:
              number: 80
---
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-ingress-nginx
  namespace: e2e-ns
spec:
  podSelector:
    matchLabels:
      app: full-app
  policyTypes:
  - Ingress
  ingress:
  - from:
    - namespaceSelector:
        matchLabels:
          kubernetes.io/metadata.name: ingress-nginx
    ports:
    - protocol: TCP
      port: 8080
```
```
kubectl apply -f e2e.yaml
```
```
# Verify
kubectl get svc full-svc -n e2e-ns
kubectl get endpoints full-svc -n e2e-ns
kubectl get ingress full-ingress -n e2e-ns
kubectl get networkpolicy -n e2e-ns -o yaml
```
