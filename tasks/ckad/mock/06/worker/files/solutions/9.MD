```
# Create overlay directory and patch file
mkdir -p /home/ubuntu/kust5/overlays/prod

# vim /home/ubuntu/kust5/overlays/prod/patch.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: cache-app
spec:
  template:
    spec:
      containers:
      - name: cache-app
        resources:
          limits:
            cpu: 200m
            memory: 256Mi
```
```
# vim /home/ubuntu/kust5/overlays/prod/kustomization.yaml

namespace: kpatch-ns
resources:
- ../../base
patches:
- path: patch.yaml
  target:
    kind: Deployment
    name: cache-app
```
```
# Apply the overlay
kubectl apply -k /home/ubuntu/kust5/overlays/prod

# Verify
kubectl get deployment cache-app -n kpatch-ns -o jsonpath='{.spec.template.spec.containers[0].resources}'
```
