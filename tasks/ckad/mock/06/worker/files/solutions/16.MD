```
# vim allow-admin-vault.yaml

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-admin-to-vault
  namespace: secure-ns
spec:
  podSelector:
    matchLabels:
      app: vault
  policyTypes:
  - Ingress
  ingress:
  - from:
    - namespaceSelector:
        matchLabels:
          team: admin
      podSelector:
        matchLabels:
          role: operator
    ports:
    - protocol: TCP
      port: 8080
```
```
# Note: namespaceSelector and podSelector in the same list item means AND (both must match)
# If they were in separate list items it would be OR logic

kubectl apply -f allow-admin-vault.yaml
```
```
# Verify - admin-pod should reach vault-app
vault_ip=$(kubectl get pod -n secure-ns -l app=vault -o jsonpath='{.items[0].status.podIP}')
kubectl exec admin-pod -n admin-ns -- wget -q --timeout=5 -O- "http://$vault_ip:8080"
```
