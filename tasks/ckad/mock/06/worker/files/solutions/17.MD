```
# vim allow-multi-src.yaml
# To allow from frontend OR monitor, use separate list items under from: (each item is OR logic)

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-frontend-and-monitor
  namespace: multi-src-ns
spec:
  podSelector:
    matchLabels:
      role: api
  policyTypes:
  - Ingress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          role: frontend
    - podSelector:
        matchLabels:
          role: monitor
    ports:
    - protocol: TCP
      port: 8080
```
```
kubectl apply -f allow-multi-src.yaml
```
