```
# vim allow-api-to-db.yaml

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-api-to-db
  namespace: tier-ns
spec:
  podSelector:
    matchLabels:
      role: db
  policyTypes:
  - Ingress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          role: api
    ports:
    - protocol: TCP
      port: 8080
```
```
kubectl apply -f allow-api-to-db.yaml
```
```
# Verify - api-pod can reach db-pod
db_ip=$(kubectl get pod db-pod -n tier-ns -o jsonpath='{.status.podIP}')
kubectl exec api-pod -n tier-ns -- wget -q --timeout=5 -O- "http://$db_ip:8080"

# worker-pod cannot reach db-pod
kubectl exec worker-pod -n tier-ns -- wget -q --timeout=5 -O- "http://$db_ip:8080"
# should fail
```
