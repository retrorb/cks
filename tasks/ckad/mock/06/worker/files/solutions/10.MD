kubectl config use-context cluster1-admin@cluster1

```
# Check current state
kubectl get all -n bg2-ns
kubectl get ingress bg-ingress -n bg2-ns -o yaml
```
```
# vim v2-deploy.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: v2-deploy
  namespace: bg2-ns
spec:
  replicas: 2
  selector:
    matchLabels:
      app: bgapp
      version: v2
  template:
    metadata:
      labels:
        app: bgapp
        version: v2
    spec:
      containers:
      - name: v2-app
        image: viktoruj/ping_pong:alpine
        ports:
        - containerPort: 80
---
apiVersion: v1
kind: Service
metadata:
  name: v2-svc
  namespace: bg2-ns
spec:
  selector:
    version: v2
  ports:
  - port: 80
    targetPort: 80
```
```
kubectl apply -f v2-deploy.yaml
```
```
# Update ingress backend to v2-svc
kubectl patch ingress bg-ingress -n bg2-ns --type=json \
  -p='[{"op":"replace","path":"/spec/rules/0/http/paths/0/backend/service/name","value":"v2-svc"}]'
```
```
# Verify
kubectl get ingress bg-ingress -n bg2-ns -o jsonpath='{.spec.rules[0].http.paths[0].backend.service.name}'
kubectl get endpoints v2-svc -n bg2-ns
```
