```
# The bug: ingress.from.podSelector.matchLabels.role = "api" (should be "gateway")
```
```
# Fix option 1: patch
kubectl patch networkpolicy allow-gateway -n fix-np-ns --type=json \
  -p='[{"op":"replace","path":"/spec/ingress/0/from/0/podSelector/matchLabels/role","value":"gateway"}]'
```
```
# Fix option 2: delete and recreate
# kubectl delete networkpolicy allow-gateway -n fix-np-ns
# vim fixed-policy.yaml

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-gateway
  namespace: fix-np-ns
spec:
  podSelector:
    matchLabels:
      role: api
  policyTypes:
  - Ingress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          role: gateway
    ports:
    - protocol: TCP
      port: 8080

# kubectl apply -f fixed-policy.yaml
```
```
# Verify - gateway-pod should reach api-service
api_ip=$(kubectl get pod api-service -n fix-np-ns -o jsonpath='{.status.podIP}')
kubectl exec gateway-pod -n fix-np-ns -- wget -q --timeout=5 -O- "http://$api_ip:8080"
```
