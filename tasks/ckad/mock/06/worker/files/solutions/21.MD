```
# vim rewrite-ingress.yaml

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: rewrite-ingress
  namespace: rewrite-ns
  annotations:
    nginx.ingress.kubernetes.io/rewrite-target: /$2
    nginx.ingress.kubernetes.io/use-regex: "true"
spec:
  ingressClassName: nginx
  rules:
  - http:
      paths:
      - path: /api(/|$)(.*)
        pathType: Prefix
        backend:
          service:
            name: api-svc21
            port:
              number: 8080
```
```
kubectl apply -f rewrite-ingress.yaml
```
```
# Verify
kubectl get ingress rewrite-ingress -n rewrite-ns -o yaml
```
