```
# Check TLS files
ls /home/ubuntu/tls/
```
```
# Create TLS secret from cert and key files
kubectl create secret tls tls-secret \
  --cert=/home/ubuntu/tls/tls.crt \
  --key=/home/ubuntu/tls/tls.key \
  -n tls-ns
```
```
# vim secure-ingress.yaml

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: secure-ingress
  namespace: tls-ns
spec:
  ingressClassName: nginx
  tls:
  - hosts:
    - secure.example.com
    secretName: tls-secret
  rules:
  - host: secure.example.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: secure-svc
            port:
              number: 80
```
```
kubectl apply -f secure-ingress.yaml
```
```
# Verify
kubectl get secret tls-secret -n tls-ns
kubectl get ingress secure-ingress -n tls-ns -o yaml
```
