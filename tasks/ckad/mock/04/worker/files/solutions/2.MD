kind: Pod

  name: init-pod
  namespace: init-ns

  initContainers:

    image: busybox
    command: ["sh", "-c", "echo 'init done' > /shared/init.txt"]
    volumeMounts:
    - name: shared-data
      mountPath: /shared
  containers:
  - name: main
    image: nginx:alpine
    volumeMounts:
    - name: shared-data
      mountPath: /shared
  volumes:
  - name: shared-data
    emptyDir: {}
