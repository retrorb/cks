```
# vim 6.yaml

apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: data-pvc
  namespace: storage-ns
spec:
  accessModes:
  - ReadWriteOnce
  resources:
    requests:
      storage: 100Mi
---
apiVersion: v1
kind: Pod
metadata:
  name: storage-pod
  namespace: storage-ns
spec:
  containers:
  - name: storage-pod
    image: nginx:alpine
    command: ["sleep", "3600"]
    volumeMounts:
    - name: data-vol
      mountPath: /data
  volumes:
  - name: data-vol
    persistentVolumeClaim:
      claimName: data-pvc
```
```
kubectl apply -f 6.yaml

kubectl get pvc data-pvc -n storage-ns
kubectl get pod storage-pod -n storage-ns
```
