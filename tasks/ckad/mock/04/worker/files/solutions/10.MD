```
# vim 10.yaml

apiVersion: v1
kind: Pod
metadata:
  name: secure-pod
  namespace: security-ns
spec:
  securityContext:
    runAsUser: 1000
    runAsGroup: 3000
    fsGroup: 2000
  containers:
  - name: secure-pod
    image: busybox
    command: ["sleep", "3600"]
    securityContext:
      allowPrivilegeEscalation: false
      readOnlyRootFilesystem: true
      capabilities:
        drop:
        - ALL
```
```
kubectl apply -f 10.yaml

kubectl get pod secure-pod -n security-ns
kubectl exec secure-pod -n security-ns -- id
```
