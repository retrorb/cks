```
kubectl edit deployment probe-app -n probe-ns
# Add under containers[0]:
#   livenessProbe:
#     httpGet:
#       path: /health
#       port: 8080
#     initialDelaySeconds: 10
#     periodSeconds: 5

# Or patch:
kubectl patch deployment probe-app -n probe-ns --type=json -p='[
  {"op":"add","path":"/spec/template/spec/containers/0/livenessProbe","value":{
    "httpGet":{"path":"/health","port":8080},
    "initialDelaySeconds":10,
    "periodSeconds":5
  }}
]'
```
```
# Verify
kubectl get deployment probe-app -n probe-ns -o jsonpath='{.spec.template.spec.containers[0].livenessProbe}'
kubectl rollout status deployment/probe-app -n probe-ns
```
