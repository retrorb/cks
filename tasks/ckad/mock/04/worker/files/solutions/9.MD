```
kubectl create ns secret-ns

kubectl create secret generic db-secret -n secret-ns \
  --from-literal=DB_USER=admin \
  --from-literal=DB_PASS=s3cr3t
```
```
# vim 9.yaml

apiVersion: v1
kind: Pod
metadata:
  name: secret-pod
  namespace: secret-ns
spec:
  containers:
  - name: secret-pod
    image: nginx:alpine
    command: ["sleep", "3600"]
    volumeMounts:
    - name: secret-vol
      mountPath: /etc/secrets
  volumes:
  - name: secret-vol
    secret:
      secretName: db-secret
```
```
kubectl apply -f 9.yaml

kubectl get pod secret-pod -n secret-ns
kubectl exec secret-pod -n secret-ns -- cat /etc/secrets/DB_USER
```
