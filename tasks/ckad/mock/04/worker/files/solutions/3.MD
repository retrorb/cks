```
# vim 3.yaml

apiVersion: v1
kind: Pod
metadata:
  name: shared-pod
  namespace: sidecar-ns
spec:
  containers:
  - name: writer
    image: busybox
    command: ["sh", "-c", "while true; do echo hello sidecar > /shared/data.txt; sleep 5; done"]
    volumeMounts:
    - name: shared-data
      mountPath: /shared
  - name: reader
    image: viktoruj/ping_pong:alpine
    command: ["tail", "-f", "/shared/data.txt"]
    volumeMounts:
    - name: shared-data
      mountPath: /shared
  volumes:
  - name: shared-data
    emptyDir: {}
```
```
kubectl apply -f 3.yaml

kubectl get pod shared-pod -n sidecar-ns
kubectl exec shared-pod -c reader -n sidecar-ns -- cat /shared/data.txt
```
