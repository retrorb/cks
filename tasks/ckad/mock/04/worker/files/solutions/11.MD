```
# vim 11.yaml

apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: deploy-reader
  namespace: rbac-ns
rules:
- apiGroups: ["apps"]
  resources: ["deployments"]
  verbs: ["get", "list", "watch"]
---
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: app-sa-binding
  namespace: rbac-ns
subjects:
- kind: ServiceAccount
  name: app-sa
  namespace: rbac-ns
roleRef:
  kind: Role
  name: deploy-reader
  apiGroup: rbac.authorization.k8s.io
---
apiVersion: v1
kind: Pod
metadata:
  name: rbac-pod
  namespace: rbac-ns
spec:
  serviceAccountName: app-sa
  containers:
  - name: rbac-pod
    image: viktoruj/cks-lab
    command: ["sleep", "60000"]
```
```
kubectl apply -f 11.yaml

kubectl get sa,role,rolebinding -n rbac-ns
kubectl get pod rbac-pod -n rbac-ns -o jsonpath='{.spec.serviceAccountName}'
```
