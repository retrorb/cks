```
kubectl create ns config-ns

kubectl create configmap app-config -n config-ns \
  --from-literal=APP_ENV=production \
  --from-literal=LOG_LEVEL=info
```
```
# vim 8.yaml

apiVersion: v1
kind: Pod
metadata:
  name: config-pod
  namespace: config-ns
spec:
  containers:
  - name: config-pod
    image: nginx:alpine
    command: ["sleep", "3600"]
    env:
    - name: APP_ENV
      valueFrom:
        configMapKeyRef:
          name: app-config
          key: APP_ENV
    volumeMounts:
    - name: config-vol
      mountPath: /etc/config
  volumes:
  - name: config-vol
    configMap:
      name: app-config
```
```
kubectl apply -f 8.yaml

kubectl get pod config-pod -n config-ns
kubectl exec config-pod -n config-ns -- sh -c 'echo $APP_ENV'
kubectl exec config-pod -n config-ns -- ls /etc/config/
```
