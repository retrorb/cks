## Task 19: Create SA, ClusterRole, CRB, Pod, Save Logs

```bash
kubectl config use-context cluster1-admin@cluster1

# Create SA
kubectl create serviceaccount log-reader-sa -n log-ns

# Create ClusterRole
cat << 'EOF' | kubectl apply -f -
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: log-reader
rules:
- apiGroups: [""]
  resources: ["pods"]
  verbs: ["get", "list"]
- apiGroups: [""]
  resources: ["pods/log"]
  verbs: ["get"]
EOF

# Create ClusterRoleBinding
kubectl create clusterrolebinding log-reader-crb \
  --clusterrole=log-reader \
  --serviceaccount=log-ns:log-reader-sa

# Create Pod
cat << 'EOF' | kubectl apply -f -
apiVersion: v1
kind: Pod
metadata:
  name: log-reader-pod
  namespace: log-ns
spec:
  serviceAccountName: log-reader-sa
  containers:
  - name: kubectl
    image: bitnami/kubectl:latest
    command: ["sleep", "3600"]
EOF

# Wait for pod to be running
kubectl wait pod log-reader-pod -n log-ns --for=condition=Ready --timeout=60s

# Run kubectl get pods and save output
kubectl exec log-reader-pod -n log-ns -- kubectl get pods -A \
  > /var/work/tests/artifacts/19/pods.txt
```

**Key concepts:**
- Pods can use the ServiceAccount token to authenticate against the K8s API
- `bitnami/kubectl` image includes kubectl CLI
- The pod uses its SA token automatically via the mounted secret
