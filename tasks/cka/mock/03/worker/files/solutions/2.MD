## Task 2: Deployment + NodePort Service

```bash
kubectl config use-context cluster1-admin@cluster1

# Create namespace
kubectl create namespace web-app

# Create deployment
kubectl create deployment frontend \
  --image=nginx:alpine \
  --replicas=2 \
  -n web-app

# Create NodePort service
kubectl expose deployment frontend \
  --name=frontend-service \
  --port=80 \
  --target-port=80 \
  --type=NodePort \
  -n web-app

# Patch the nodePort to 30080
kubectl patch svc frontend-service -n web-app \
  --type='json' \
  -p='[{"op":"replace","path":"/spec/ports/0/nodePort","value":30080}]'
```

Or use a complete manifest:

```yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: frontend
  namespace: web-app
spec:
  replicas: 2
  selector:
    matchLabels:
      app: frontend
  template:
    metadata:
      labels:
        app: frontend
    spec:
      containers:
      - name: frontend
        image: nginx:alpine
        ports:
        - containerPort: 80
---
apiVersion: v1
kind: Service
metadata:
  name: frontend-service
  namespace: web-app
spec:
  type: NodePort
  selector:
    app: frontend
  ports:
  - port: 80
    targetPort: 80
    nodePort: 30080
```

**Key concepts:**
- NodePort exposes the service on each node's IP at a static port (30080)
- NodePort range: 30000–32767
